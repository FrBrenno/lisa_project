158void HomeFrame::updateImage(wxTimerEvent& event) {
159 this->imageControl->Freeze();
160 if (!controller->isWfsConnected()) {
161 this->stopPreview();
162 return;
163 }
164
165 wxImage* image = new wxImage();
166 for (BaseController* listener : this->listeners)
167 {
168 if (ImageController* imgController = dynamic_cast<ImageController*>(listener))
169 {
170 imgController->takeImage();
171
172 if (imgController->hasError() != 0)
173 {
174 this->stopPreview();
175 return;
176 }
177 image = imgController->getImage();
178 }
179 }
180 this->setImage(image);
181 this->imageControl->Thaw();
182}

193void HomeFrame::resizeImage(wxImage* image)
194{
195 // if the image is already in the right size, do not resize
196 if (image->GetWidth() == imageControl->GetSize().GetWidth() && image->GetHeight() == imageControl->GetSize().GetHeight())
197 return;
198
199 int width = image->GetWidth();
200 int height = image->GetHeight();
201 int newWidth = imageControl->GetSize().GetWidth();
202 int newHeight = imageControl->GetSize().GetHeight();
203 if (width > height) {
204 newHeight = (int)(newWidth * ((float)height / (float)width));
205 }
206 else {
207 newWidth = (int)(newHeight * ((float)width / (float)height));
208 }
209 *image = image->Rescale(newWidth, newHeight);
210 return;
211}
