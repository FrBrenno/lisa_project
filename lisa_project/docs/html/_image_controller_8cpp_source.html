112void ImageController::convertGrayscaleToRGB(const unsigned char* grayscaleBuffer, int width, int height, unsigned char* rgbBuffer) {
113 // Assuming your grayscaleBuffer contains width * height pixel values
114 // Iterate through each pixel in the grayscale image
115 for (int i = 0; i < width * height; ++i) {
116 // Grayscale pixel value (0 to 255)
117 unsigned char grayValue = grayscaleBuffer[i];
118
119 // Copy the grayscale value to all three channels (RGB)
120 rgbBuffer[3 * i] = grayValue; // Red channel
121 rgbBuffer[3 * i + 1] = grayValue; // Green channel
122 rgbBuffer[3 * i + 2] = grayValue; // Blue channel
123 }
