229 ViSession handle = *this->selectedInstrument->getHandle();

232 // If device is a WFS20

234 ViInt32* spotsX = this->selectedInstrument->getSpotsX();
235 ViInt32* spotsY = this->selectedInstrument->getSpotsY();
236 if (err = WFS_ConfigureCam(handle,
237 SAMPLE_PIXEL_FORMAT,
238 SAMPLE_CAMERA_RESOL_WFS20,
239 spotsX,
240 spotsY))
241 {
242 this->handleError(err, "Not able to configure camera");
243 return;
244 }
245 }
246 else {
247 // Other devices are not compatible with this software for now
248 this->handleError(-1, "The device selected is not a WFS20. Others devices are not compatible.");
249 }
250
251 // WFS Internal Reference Plane Setup
252 if (err = WFS_SetReferencePlane(handle, SAMPLE_REF_PLANE)) {
253 this->handleError(err, "Not able to set reference plane");
254 return;
